  \begin{example}\label{mai:exam112}
    \begin{equation}\label{mai:exam016_001}
      \int{xe^x\dd{x}}
    \end{equation}
    Užijeme metodu per partes 
    \begin{equation*}
      \left[
        \begin{array}{cc}
          u=x   & dv=e^x \\
          du=dx & v=e^x
        \end{array}
      \right] 
    \end{equation*}

    \begin{equation*}
      \int{xe^xdx} = xe^x-\int{e^x\dd{x}} = xe^x - e^x+ c
    \end{equation*}
  \end{example}
\end{mdframed}
