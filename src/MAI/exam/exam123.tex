  \begin{example}\label{mai:exam123}
    \begin{equation*}
      \int\frac{x}{x^2+1}\dd{x}
    \end{equation*}
    Použijeme substituci
    \begin{equation*}
      \left[
        \begin{array}{c} 
          x^2 + 1 = t  \Rightarrow 2x\dd{x} = \dd{t}        \\ 
              x\dd{x} = \displaystyle{\frac{\dd{t}}{2}}
        \end{array} 
      \right]        
    \end{equation*}
    \begin{equation*}
      \frac{1}{2}\int\frac{dt}{t} = \frac{1}{2}\ln\abs{t} =\frac{1}{2}\ln\abs{1+x^2}+ c
    \end{equation*}
  \end{example}
\end{mdframed}
